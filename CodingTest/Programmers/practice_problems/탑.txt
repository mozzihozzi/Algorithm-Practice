#include <iostream>
#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> heights) {
	vector<int> answer;


	for (int i = 0; i < heights.size(); i++){
		int idx = 0;
		for (int j = i - 1; j >= 0; j--){
			if (heights[i] < heights[j]){
				idx = j + 1;
				break;
			}
		}

		answer.push_back(idx);
	}

	return answer;
}

int main(void){
	vector<int> heights;

	solution(heights);

	return 0;
}
