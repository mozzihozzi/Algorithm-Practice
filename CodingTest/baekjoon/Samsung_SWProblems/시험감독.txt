#include <iostream>

using namespace std;

int N, B, C;
int arr[1000000];

unsigned long long simulate() {
	unsigned long long result = 0;
	unsigned long long b, c;
	for (int i = 0; i < N; i++) {
		arr[i] -= B;		
		b = arr[i] / C;
		c = arr[i] % C;

		if (arr[i] <= 0) {	//총감독관만 으로도 충분한 경우
			result += 1;
		}
		else if (c == 0) {
			result += b + 1;
		}
		else {
			result += b + 2;
		}
	}

	return result;
}


int main(void) {

	scanf("%d", &N);
	for (int i = 0; i < N; i++) {
		scanf("%d", &arr[i]);
	}
	scanf("%d", &B);
	scanf("%d", &C);
	
	printf("%lld\n",simulate());

	return 0;
}
